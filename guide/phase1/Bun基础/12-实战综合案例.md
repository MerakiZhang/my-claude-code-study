## 12.1 项目需求：智能配置文件管理器

我们要构建一个叫 `configman` 的工具，功能包括：

1. **运行时自适应**：在 Bun 下用高性能 API，在 Node.js 下回退到标准 API
2. **特性标记系统**：通过编译时特性开关控制功能
3. **多格式配置支持**：支持 JSON、YAML、JSONL 格式
4. **配置校验**：使用语义版本控制检查配置版本兼容性
5. **命令查找**：检查系统中必需工具是否可用
6. **安全模式**：bundled 模式下启用额外的安全设置
7. **内存监控**：可选的内存使用监控功能
8. **堆快照诊断**：在调试模式下生成堆快照

import type { RuntimeInfo } from './types';

export function isRunningWithBun(): boolean {
  return process.versions.bun !== undefined;

export function isInBundledMode(): boolean {

### `src/hash.ts` —— 哈希工具

```typescript
// src/hash.ts

export function hashContent(content: string): string {
  if (typeof Bun !== 'undefined') {
    return Bun.hash(content).toString(36);
  }

  // Node.js 回退：djb2 哈希算法
  let hash = 5381;
  for (let i = 0; i < content.length; i++) {
    hash = ((hash << 5) + hash + content.charCodeAt(i)) & 0xffffffff;
  }
  return hash.toString(36);
}

export function hashPair(a: string, b: string): string {
  if (typeof Bun !== 'undefined') {
    return Bun.hash(b, Bun.hash(a)).toString(36);
  }

  // 回退：简单组合哈希
  return hashContent(a + '\0' + b);
}
```

### `src/semver.ts` —— 版本比较封装

```typescript
// src/semver.ts

export function satisfies(version: string, range: string): boolean {
  if (typeof Bun !== 'undefined' && Bun.semver) {
    return Bun.semver.satisfies(version, range);
  }

  // 简化回退：只支持 >=X.Y.Z 格式
  const match = range.match(/^>=\s*(\d+)\.(\d+)\.(\d+)$/);
  if (!match) {
    throw new Error(`不支持的版本范围格式: ${range}`);
  }

  const [_, reqMajor, reqMinor, reqPatch] = match.map(Number);
  const [major, minor, patch] = version.split('.').map(Number);

  if (major > reqMajor) return true;
  if (major < reqMajor) return false;
  if (minor > reqMinor) return true;
  if (minor < reqMinor) return false;
  return patch >= reqPatch;
}
```

### `src/which.ts` —— 命令查找封装

```typescript
// src/which.ts

const bunWhich = typeof Bun !== 'undefined' && typeof Bun.which === 'function'
  ? Bun.which
  : null;

export function which(command: string): string | null {
  if (bunWhich) {
    return bunWhich(command);
  }

  // Node.js 回退
  const { execSync } = require('child_process');
  const platformCmd = process.platform === 'win32' ? 'where' : 'which';
  try {
    return execSync(`${platformCmd} ${command}`, { encoding: 'utf-8' }).trim().split('\n')[0] || null;
  } catch {
    return null;
  }
}
```

### `src/config.ts` —— 配置管理核心

```typescript
// src/config.ts
import type { AppConfig } from './types';
import { parseYaml } from './yaml';
import { hashContent } from './hash';
import { satisfies } from './semver';

export async function loadConfig(path: string): Promise<AppConfig> {
  const file = Bun.file(path);
  if (!(await file.exists())) {
    throw new Error(`配置文件不存在: ${path}`);
  }

  const content = await file.text();
  const hash = hashContent(content);
  console.log(`[配置] 加载 ${path} (hash: ${hash})`);

  let raw: unknown;
  if (path.endsWith('.json')) {
    raw = JSON.parse(content);
  } else if (path.endsWith('.yaml') || path.endsWith('.yml')) {
    raw = parseYaml(content);
  } else {
    throw new Error(`不支持的配置文件格式: ${path}`);
  }

  const config = raw as AppConfig;

  // 版本检查
  const runtimeVersion = process.versions.bun || process.versions.node || '0.0.0';
  if (!satisfies(runtimeVersion, config.minRuntimeVersion)) {
    throw new Error(
      `运行时版本 ${runtimeVersion} 不满足最低要求 ${config.minRuntimeVersion}`
    );
  }

  return config;
}

export async function saveConfig(path: string, config: AppConfig): Promise<void> {
  const content = JSON.stringify(config, null, 2);
  await Bun.write(path, content);
  console.log(`[配置] 已保存到 ${path}`);
}
```

### `src/security.ts` —— 安全设置

```typescript
// src/security.ts
import { isInBundledMode } from './runtime';

export function applySecurityHardening(): void {
  if (!isInBundledMode()) {
    console.log('[安全] 非 bundled 模式，跳过安全加固');
    return;
  }

  console.log('[安全] 应用 bundled 模式安全加固');

  // Linux 下设置进程不可转储
  if (process.platform === 'linux') {
    try {
      const ffi = require('bun:ffi') as typeof import('bun:ffi');
      const lib = ffi.dlopen('libc.so.6', {
        prctl: {
          args: [ffi.FFIType.int, ffi.FFIType.int],
          returns: ffi.FFIType.int,
        },
      });
      lib.symbols.prctl(4, 0); // PR_SET_DUMPABLE = 4
      console.log('[安全] 已禁用进程内存转储');
    } catch (err) {
      console.warn('[安全] 无法设置进程转储保护:', err);
    }
  }

  // 设置安全的环境变量
  process.env.NoDefaultCurrentDirectoryInExePath = '1';
  console.log('[安全] 已设置 PATH 安全策略');
}
```

### `src/monitor.ts` —— 内存监控

```typescript
// src/monitor.ts

export class MemoryMonitor {
  // Timer 类型来自 @types/bun，如果没有安装可以用 ReturnType<typeof setInterval>
  private timer: ReturnType<typeof setInterval> | null = null;
  private samples: Array<{ time: number; heapUsed: number; rss: number }> = [];

  start(intervalMs: number = 5000): void {
    if (this.timer) return;

    this.timer = setInterval(() => {
      const usage = process.memoryUsage();
      this.samples.push({
        time: Date.now(),
        heapUsed: Math.round(usage.heapUsed / 1024 / 1024),
        rss: Math.round(usage.rss / 1024 / 1024),
      });
    }, intervalMs);

    // 不阻止进程退出
    if ('unref' in this.timer) {
      this.timer.unref();
    }

    console.log(`[监控] 内存监控已启动（每 ${intervalMs}ms）`);
  }

  stop(): void {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
      console.log('[监控] 内存监控已停止');
    }
  }

  report(): void {
    if (this.samples.length === 0) {
      console.log('[监控] 没有采样数据');
      return;
    }

    const heaps = this.samples.map(s => s.heapUsed);
    const rsses = this.samples.map(s => s.rss);

    console.log('\n=== 内存监控报告 ===');
    console.log(`采样次数: ${this.samples.length}`);
    console.log(`Heap: 最小 ${Math.min(...heaps)}MB | 最大 ${Math.max(...heaps)}MB | 平均 ${Math.round(heaps.reduce((a, b) => a + b) / heaps.length)}MB`);
    console.log(`RSS:  最小 ${Math.min(...rsses)}MB | 最大 ${Math.max(...rsses)}MB | 平均 ${Math.round(rsses.reduce((a, b) => a + b) / rsses.length)}MB`);
  }
}
```

### `src/diagnostics.ts` —— 诊断工具

```typescript
// src/diagnostics.ts
import type { DiagnosticResult } from './types';
import { getRuntimeInfo } from './runtime';
import { which } from './which';

export async function runDiagnostics(): Promise<DiagnosticResult[]> {
  const results: DiagnosticResult[] = [];
  const runtime = getRuntimeInfo();

  // 运行时检测
  results.push({
    category: '运行时',
    check: 'Bun 运行时',
    status: runtime.isBun ? 'ok' : 'warn',
    message: runtime.isBun ? `Bun ${runtime.bunVersion}` : '使用 Node.js 回退模式',
  });

  results.push({
    category: '运行时',
    check: 'Bundled 模式',
    status: runtime.isBundled ? 'ok' : 'ok',
    message: runtime.isBundled ? '独立可执行文件' : '源码运行模式',
  });

  // 必需工具检查
  const requiredTools = ['git', 'node'];
  for (const tool of requiredTools) {
    const path = which(tool);
    results.push({
      category: '工具',
      check: tool,
      status: path ? 'ok' : 'error',
      message: path || '未找到',
    });
  }

  // 可选工具检查
  const optionalTools = ['rg', 'gh', 'docker'];
  for (const tool of optionalTools) {
    const path = which(tool);
    results.push({
      category: '工具',
      check: tool,
      status: path ? 'ok' : 'warn',
      message: path || '未找到（可选）',
    });
  }

  return results;
}

export function printDiagnostics(results: DiagnosticResult[]): void {
  console.log('\n=== 系统诊断报告 ===\n');

  const categories = [...new Set(results.map(r => r.category))];
  for (const category of categories) {
    console.log(`[${category}]`);
    for (const r of results.filter(x => x.category === category)) {
      const icon = r.status === 'ok' ? '✅' : r.status === 'warn' ? '⚠️' : '❌';
      console.log(`  ${icon} ${r.check}: ${r.message}`);
    }
    console.log('');
  }
}
```

### `src/main.ts` —— 入口文件

```typescript
// src/main.ts
import { Command } from 'commander';
import { feature } from './features';
import { getRuntimeInfo } from './runtime';
import { loadConfig, saveConfig } from './config';
import { applySecurityHardening } from './security';
import { MemoryMonitor } from './monitor';
import { runDiagnostics, printDiagnostics } from './diagnostics';
import type { AppConfig } from './types';

const program = new Command();

program
  .name('configman')
  .description('智能配置文件管理器')
  .version('1.0.0');

// 运行时信息命令
program
  .command('info')
  .description('显示运行时信息')
  .action(() => {
    const info = getRuntimeInfo();
    console.log('=== 运行时信息 ===');
    console.log(`Bun: ${info.isBun ? `✅ ${info.bunVersion}` : '❌'}`);
    console.log(`Bundled: ${info.isBundled ? '✅' : '❌'}`);
    console.log(`Node.js: ${info.nodeVersion || 'N/A'}`);
    console.log(`平台: ${info.platform}`);
    console.log(`可执行文件: ${info.execPath}`);
    console.log(`脚本路径: ${info.scriptPath || 'N/A (bundled)'}`);

    console.log('\n=== 启用特性 ===');
    const { listEnabledFeatures } = require('./features');
    for (const f of listEnabledFeatures()) {
      console.log(`  • ${f}`);
    }
  });

// 加载配置命令
program
  .command('load <path>')
  .description('加载配置文件')
  .action(async (path: string) => {
    try {
      const config = await loadConfig(path);
      console.log('\n=== 配置内容 ===');
      console.log(JSON.stringify(config, null, 2));
    } catch (err) {
      console.error(`错误: ${err}`);
      process.exit(1);
    }
  });

// 诊断命令
program
  .command('doctor')
  .description('运行系统诊断')
  .action(async () => {
    const results = await runDiagnostics();
    printDiagnostics(results);
  });

// 安全加固命令（bundled 模式）
if (feature('SECURITY_HARDENING')) {
  program
    .command('harden')
    .description('应用安全加固')
    .action(() => {
      applySecurityHardening();
    });
}

// 内存监控命令
if (feature('MEMORY_MONITOR')) {
  program
    .command('monitor')
    .description('监控内存使用')
    .option('-d, --duration <seconds>', '监控持续时间', '10')
    .action(async (options: { duration: string }) => {
      const duration = parseInt(options.duration, 10) * 1000;
      const monitor = new MemoryMonitor();
      monitor.start(1000);

      console.log(`开始监控 ${options.duration} 秒...`);
      await new Promise(resolve => setTimeout(resolve, duration));

      monitor.stop();
      monitor.report();
    });
}

// 堆快照命令
if (feature('HEAP_SNAPSHOT')) {
  program
    .command('snapshot')
    .description('生成堆快照')
    .option('-o, --output <path>', '输出路径', `heap-${Date.now()}.heapsnapshot`)
    .action(async (options: { output: string }) => {
      if (typeof Bun === 'undefined' || !Bun.generateHeapSnapshot) {
        console.error('需要 Bun 运行时支持堆快照');
        process.exit(1);
      }

      Bun.gc(true);
      const snapshot = Bun.generateHeapSnapshot('v8', 'arraybuffer');
      await Bun.write(options.output, snapshot);
      console.log(`堆快照已保存: ${options.output}`);
    });
}

// 默认命令：显示帮助
if (process.argv.length <= 2) {
  program.help();
}

program.parse();
```
